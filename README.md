# git-config

```
[alias]
        ci = commit
        ck = checkout
        br = branch
        st = status
        re = reset
        rv = revert
        go = checkout -b
        l = log HEAD --stat --graph --name-status --pretty=format:'----------------------------%n%C(yellow)%h%C(white)%d %C(blue)%an, %Creset%ar%n%n%Cgreen%s%n'
        reci = commit --amend
        unadd = reset HEAD
        sm = merge --squash
        subpull = submodule foreach git pull origin master

```
